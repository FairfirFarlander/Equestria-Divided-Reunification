	if={limit={NOT={has_tech=melee_weapons1}}
		add_tech_bonus={
			name=musket_weaponry
			bonus=0.75	uses=2
			category=infantry_weapons
		}
		add_tech_bonus={
			name=renaissance_artillery
			bonus=0.75	uses=1
			category=artillery
		}
	}
	if={limit={AND={has_tech=melee_weapons1 NOT={has_tech=pre_gw_artillery}}}
		army_experience=5
		add_tech_bonus={
			name=rifle_weaponry
			bonus=0.65	uses=1
			category=infantry_weapons
		}
		add_tech_bonus={
			name=renaissance_artillery
			bonus=0.55	uses=1
			category=artillery
		}
	}
	if={limit={AND={has_tech=melee_weapons1 has_tech=pre_gw_artillery NOT={has_tech=infantry_weapons}}}
		army_experience=10
		add_tech_bonus={
			name=rifle_weaponry
			bonus=0.55	uses=1
			category=infantry_weapons
		}
	}
	if={limit={has_tech=infantry_weapons}
		army_experience=15
	}
}
