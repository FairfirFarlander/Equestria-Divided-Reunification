	if={limit={has_dlc="Man the Guns"}
		custom_effect_tooltip=EDR_N01_tooltipMTG}
	if={limit={not={has_dlc="Man the Guns"}}
		custom_effect_tooltip=EDR_N01_tooltipNON-MTG}

		add_tech_bonus={
			name=battery_bonus
			bonus=0.75	uses=1

		}

		add_tech_bonus={
			name=transport_bonus
			bonus=0.75	uses=1
			technology=mtg_transport
		}
	}

		custom_effect_tooltip=EDR_N01_tooltipB
		add_tech_bonus={
			name=transport_bonus
			bonus=0.75	uses=1
			technology=transport
		}
	}
}
